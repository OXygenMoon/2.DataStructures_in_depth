 21 #define VAL(n)  (n ? n->val : 0)
 22 #define SIZE(n) (n ? n->size : 0)
 23 #define L(n)    (n ? n->lchild : nullptr)
 24 #define R(n)    (n ? n->rchild : nullptr)
 25 
 26 
 27 typedef struct Node{
 28     int val, size;
 29     Node *lchild, *rchild;
 30 }Node;
 31 
 32 void update_size(Node *root){
 33     root->size = SIZE(L(root)) + SIZE(R(root)) + 1;
 34     return;
 35 }
 36 
 37 Node * getNewNode(int target){
 38     Node * p = (Node *)malloc(sizeof(Node));
 39     p->val = target;
 40     p->size = 1;
 41     p->lchild = p->rchild = nullptr;
 42     return p;
 43 }
 44 
 45 int search(Node *root, int target){
 46     //寻找 target 是否存在root这棵树中
 47     if(root == nullptr)
 48         return 0;
 49     if(root->val == target)
 50         return 1;
 51     if(root->val > target)
 52         return search(root->lchild, target);
 53     else
 54         return search(root->rchild, target);
 55 }
 56 
 57 int search_k(Node *root, int k){
 58     //寻找 root这棵树中第k大的数
 59     if(root == nullptr || root->size < k)
 60         return -1;
 61     if(SIZE(L(root)) == k - 1)
 62         return root->val;
 63     if(k <= SIZE(L(root)))
 64         return search_k(root->lchild, k);
 65     else
 66         return search_k(root->rchild, k - SIZE(L(root)) - 1);
 67 }
 68 
 69 Node * insert(Node *root, int target){
 70     if(root == nullptr)
 71         return getNewNode(target);
 72     if(root->val == target)
 73         return root;
 74     if(root->val > target)
 75         root->lchild = insert(root->lchild, target);
 76     else
 77         root->rchild = insert(root->rchild, target);
 78     update_size(root);
 79     return root;
 80 }
 81 
 82 Node * predecessor(Node *root){
 83     Node * temp = root->lchild;
 84     while(temp->rchild != nullptr)
 85         temp = temp->rchild;
 86     return temp;
 87 }
 88 
 89 Node * erase(Node *root, int target){
 90     if(root == nullptr) return nullptr;
 91     if(root->val > target)
 92         root->lchild = erase(root->lchild, target);
 93     else if(root->val < target)
 94         root->rchild = erase(root->rchild, target);
 95     else{
 96         if(root->lchild == nullptr || root->rchild == nullptr){
 97             Node *temp = root->lchild ? root->lchild : root->rchild;
 98             free(root);
 99             root = temp;
100         } else {
101             Node *temp = predecessor(root);
102             root->val = temp->val;
103             root->lchild = erase(root->lchild, root->val);
104         }
105     }
106     update_size(root);
107     return root;
108 }
109 
110 void print(Node *root){
111     if(root == nullptr) return;
112     printf("(%d[%d], %d, %d)\n",
113             VAL(root),SIZE(root),
114             VAL(L(root)),
115             VAL(R(root))
116             );
117     return;
118 }
119 
120 void output(Node *root){
121     if(root == nullptr) return;
122     output(root->lchild);
123     print(root);
124     output(root->rchild);
125     return;
126 }
127 
128 void output_k(Node *root, int k){
129     //输出root这棵树前k个数
130     if(root == nullptr) return;
131     if(SIZE(L(root)) >= k)
132         output_k(root->lchild, k);
133     else{
134         output(root->lchild);
135         print(root);
136         output_k(root->rchild, k - SIZE(L(root)) - 1);
137     }
138     return;
139 }
